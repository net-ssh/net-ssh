def initialize(info)
  if info[:type] == 'keyboard-interactive'
    $stdout.puts(info[:name]) unless info[:name].empty?
    $stdout.puts(info[:instruction]) unless info[:instruction].empty?
  end

def ask(prompt, echo=true)
  $stdout.print(prompt)
  $stdout.flush
  ret = $stdin.noecho(&:gets).chomp
  $stdout.print("\n")
  ret
end
